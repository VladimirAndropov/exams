#!/bin/bash

# Проверяем, что передан аргумент
if [ -z "$1" ]; then
  echo "Ошибка: не передано имя каталога."
  echo "Использование: $0 <имя_каталога>"
  exit 1
fi

# Путь к каталогу
DIR="/home/$1"

# Проверяем, существует ли каталог
if [ -d "$DIR" ]; then
  # Удаляем каталог без необходимости подтверждения
  rm -rf "$DIR"
  echo "Каталог '$DIR' успешно удалён."
else
  echo "Ошибка: Каталог '$DIR' не существует."
  exit 1
fi
