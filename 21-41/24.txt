#!/bin/bash

# Проверяем, что передан аргумент
if [ -z "$1" ]; then
  echo "Ошибка: не передано имя пользователя."
  echo "Использование: $0 <имя_пользователя>"
  exit 1
fi

# Имя пользователя
USER="$1"

# Получаем строку из /etc/passwd для указанного пользователя
USER_INFO=$(grep "^$USER:" /etc/passwd)

# Проверяем, найден ли пользователь
if [ -z "$USER_INFO" ]; then
  echo "Пользователь '$USER' не найден."
  exit 1
fi

# Извлекаем командный интерпретатор из строки
SHELL=$(echo "$USER_INFO" | cut -d: -f7)

# Выводим командный интерпретатор
echo "Командный интерпретатор пользователя '$USER': $SHELL"
