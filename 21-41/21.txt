#!/bin/bash

# Проверка, что передан ровно один аргумент
if [ "$#" -ne 1 ]; then
    echo "Использование: $0 <имя_файла>"
    exit 1
fi

# Аргумент скрипта
file_name=$1

# Проверка на случай, если передано "*"
if [ "$file_name" == "*" ]; then
    echo "Ошибка: недопустимое имя файла '*'"
    exit 1
fi

# Путь к файлу в домашней директории
file_path="/home/$file_name"

# Удаление файла без подтверждения
# Используем -- чтобы избежать проблем с именами, начинающимися с "-"
rm -f -- "$file_path"

# Проверка успешности удаления
if [ $? -eq 0 ]; then
    echo "Файл '$file_name' успешно удалён."
else
    echo "Ошибка при удалении файла '$file_name'. Возможно, файл не существует."
fi
