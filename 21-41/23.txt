#!/bin/bash

# Имя пользователя, для которого нужно найти домашнюю директорию
USER="www-data"

# Получаем строку из /etc/passwd для указанного пользователя
USER_INFO=$(grep "^$USER:" /etc/passwd)

# Проверяем, найден ли пользователь
if [ -z "$USER_INFO" ]; then
  echo "Пользователь '$USER' не найден."
  exit 1
fi

# Извлекаем домашнюю директорию из строки
HOME_DIR=$(echo "$USER_INFO" | cut -d: -f6)

# Выводим домашнюю директорию
echo "Домашняя директория пользователя '$USER': $HOME_DIR"
